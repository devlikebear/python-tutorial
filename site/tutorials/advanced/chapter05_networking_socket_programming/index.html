class SocketPool:
    """ì†Œì¼“ í’€ ê´€ë¦¬ì"""

    def __init__(self, max_size: int = 10):
        self.max_size = max_size
        self.available_sockets = []
        self.in_use_sockets = set()

        self.stats = {
            'created': 0,
            'reused': 0,
            'closed': 0
        }

    def get_socket(self, socket_type: SocketType = SocketType.TCP) -> AdvancedSocket:
        """ì†Œì¼“ íšë“"""

            # ì‚¬ìš© ê°€ëŠ¥í•œ ì†Œì¼“ í™•ì¸

                    self.in_use_sockets.add(sock)

                    return sock

            # ìƒˆ ì†Œì¼“ ìƒì„±

            self.in_use_sockets.add(new_socket)
            self.stats['created'] += 1
            print(f"   ìƒˆ ì†Œì¼“ ìƒì„±: {socket_type.name}")
            return new_socket

    def return_socket(self, sock: AdvancedSocket):
        """ì†Œì¼“ ë°˜í™˜"""
        with self.lock:
            if sock in self.in_use_sockets:
                self.in_use_sockets.remove(sock)

                    # ì†Œì¼“ ìƒíƒœ ì´ˆê¸°í™”
                    sock.buffer = b''
                    sock.state = ConnectionState.DISCONNECTED

                else:
                    # í’€ì´ ê°€ë“ ì°¬ ê²½ìš° ì†Œì¼“ ë‹«ê¸°
                    sock.close()
                    self.stats['closed'] += 1
                    print("   ì†Œì¼“ í’€ ê°€ë“ì°¸, ì†Œì¼“ ë‹«ê¸°")

    def close_all(self):
        """ëª¨ë“  ì†Œì¼“ ë‹«ê¸°"""

            all_sockets = list(self.available_sockets) + list(self.in_use_sockets)

            self.available_sockets.clear()
            self.in_use_sockets.clear()
            print("   ëª¨ë“  ì†Œì¼“ ë‹«ê¸° ì™„ë£Œ")

            }

                except socket.error as e:
                    if self.running:
                        print(f"   UDP ìˆ˜ì‹  ì˜¤ë¥˜: {e}")
                        self.stats['errors'] += 1

        except Exception as e:

class AsyncNetworkMessage:

        self.payload = payload

        self.timestamp = time.time()

    def to_dict(self) -> Dict[str, Any]:
        """ë”•ì…”ë„ˆë¦¬ë¡œ ë³€í™˜"""
        return {
            'type': self.msg_type,
            'payload': self.payload,
            'sender_id': self.sender_id,
            'correlation_id': self.correlation_id,
            'timestamp': self.timestamp

    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> 'AsyncNetworkMessage':
        """ë”•ì…”ë„ˆë¦¬ì—ì„œ ìƒì„±"""
        return cls(
            msg_type=data['type'],

            correlation_id=data.get('correlation_id')
        )

    async def to_bytes(self) -> bytes:
        """ë¹„ë™ê¸° ì§ë ¬í™”"""
        data = json.dumps(self.to_dict()).encode('utf-8')
        header = struct.pack('>I', len(data))
        return header + data

        """StreamReaderì—ì„œ ë©”ì‹œì§€ ì½ê¸°"""

        except (asyncio.IncompleteReadError, json.JSONDecodeError) as e:
            return None

    def __init__(self):

        """ë©”ì‹œì§€ ì „ì†¡ ë° ì‘ë‹µ ëŒ€ê¸°"""

            # ì‘ë‹µ ëŒ€ê¸°ë¥¼ ìœ„í•œ Future ìƒì„±

            self.writer.write(data)
            await self.writer.drain()

            self.stats['messages_sent'] += 1
            print(f"   ë¹„ë™ê¸° ë©”ì‹œì§€ ì „ì†¡: {message.msg_type}")

                    response = await asyncio.wait_for(future, timeout=5.0)
                    return response
                except asyncio.TimeoutError:
                    print(f"   ì‘ë‹µ íƒ€ì„ì•„ì›ƒ: {message.correlation_id}")
                    return None
                finally:
                    self.response_futures.pop(message.correlation_id, None)

            return None

        except Exception as e:
            print(f"   ë¹„ë™ê¸° ì „ì†¡ ì‹¤íŒ¨: {e}")
            self.stats['errors'] += 1
            return None

        try:

                if message is None:

                print(f"   ë¹„ë™ê¸° ë©”ì‹œì§€ ìˆ˜ì‹ : {message.msg_type}")

                    future = self.response_futures[message.correlation_id]

        except Exception as e:
            print(f"   ìˆ˜ì‹  ë£¨í”„ ì˜¤ë¥˜: {e}")
            self.stats['errors'] += 1
        finally:
            self.connected = False

        return {

            'connected': self.connected,
            'pending_responses': len(self.response_futures)

class HTTPRequest:
    """HTTP ìš”ì²­ íŒŒì„œ"""

    def __init__(self, raw_request: str):
        self.raw = raw_request
        self.method = ""
        self.path = ""
        self.version = ""
        self.headers = {}
        self.query_params = {}
        self.body = ""
        self.cookies = {}

        self._parse_request()

    def _parse_request(self):
        """ìš”ì²­ íŒŒì‹±"""
        if not self.raw.strip():
            return

        lines = self.raw.split('\r\n')
        if not lines:
            return

        # ì²« ë²ˆì§¸ ì¤„: ë©”ì„œë“œ, ê²½ë¡œ, ë²„ì „
        request_line = lines[0].split(' ')
        if len(request_line) >= 3:
            self.method = request_line[0]
            full_path = request_line[1]
            self.version = request_line[2]

            # ê²½ë¡œì™€ ì¿¼ë¦¬ íŒŒë¼ë¯¸í„° ë¶„ë¦¬
            if '?' in full_path:
                self.path, query_string = full_path.split('?', 1)
                self.query_params = urllib.parse.parse_qs(query_string)
            else:
                self.path = full_path

        # í—¤ë” íŒŒì‹±
        header_end = len(lines)
        for i, line in enumerate(lines[1:], 1):
            if line == '':
                header_end = i
                break

            if ':' in line:
                key, value = line.split(':', 1)
                self.headers[key.strip().lower()] = value.strip()

        # ì¿ í‚¤ íŒŒì‹±
        if 'cookie' in self.headers:
            cookie_header = self.headers['cookie']
            for cookie in cookie_header.split(';'):
                if '=' in cookie:
                    key, value = cookie.split('=', 1)
                    self.cookies[key.strip()] = value.strip()

        # ë°”ë”” íŒŒì‹±
        if header_end < len(lines):
            self.body = '\r\n'.join(lines[header_end + 1:])

    def get_header(self, name: str, default: str = None) -> str:
        """í—¤ë” ê°’ ê°€ì ¸ì˜¤ê¸°"""
        return self.headers.get(name.lower(), default)

    def get_query_param(self, name: str, default: str = None) -> str:
        """ì¿¼ë¦¬ íŒŒë¼ë¯¸í„° ê°€ì ¸ì˜¤ê¸°"""
        values = self.query_params.get(name, [default])
        return values[0] if values else default

    def get_cookie(self, name: str, default: str = None) -> str:
        """ì¿ í‚¤ ê°’ ê°€ì ¸ì˜¤ê¸°"""
        return self.cookies.get(name, default)

class HTTPRouter:
    """HTTP ë¼ìš°í„°"""

    def __init__(self):
        self.routes = {}
        self.middleware = []

    def add_route(self, method: str, path: str, handler: Callable):
        """ë¼ìš°íŠ¸ ì¶”ê°€"""
        key = f"{method.upper()}:{path}"
        self.routes[key] = handler
        print(f"   ë¼ìš°íŠ¸ ë“±ë¡: {method.upper()} {path}")

    def get(self, path: str):
        """GET ë¼ìš°íŠ¸ ë°ì½”ë ˆì´í„°"""
        def decorator(handler):
            self.add_route('GET', path, handler)
            return handler
        return decorator

    def post(self, path: str):
        """POST ë¼ìš°íŠ¸ ë°ì½”ë ˆì´í„°"""
        def decorator(handler):
            self.add_route('POST', path, handler)
            return handler
        return decorator

    def add_middleware(self, middleware: Callable):
        """ë¯¸ë“¤ì›¨ì–´ ì¶”ê°€"""
        self.middleware.append(middleware)
        print(f"   ë¯¸ë“¤ì›¨ì–´ ë“±ë¡: {middleware.__name__}")

    def find_handler(self, method: str, path: str) -> Optional[Callable]:
        """í•¸ë“¤ëŸ¬ ì°¾ê¸°"""
        # ì •í™•í•œ ë§¤ì¹˜ ë¨¼ì € í™•ì¸
        exact_key = f"{method.upper()}:{path}"
        if exact_key in self.routes:
            return self.routes[exact_key]

        # íŒ¨í„´ ë§¤ì¹­ (ê°„ë‹¨í•œ êµ¬í˜„)
        for route_key, handler in self.routes.items():
            route_method, route_path = route_key.split(':', 1)
            if route_method == method.upper():
                # ì™€ì¼ë“œì¹´ë“œ íŒ¨í„´ ì²˜ë¦¬
                if route_path.endswith('/*') and path.startswith(route_path[:-1]):
                    return handler

        return None

    def apply_middleware(self, request: HTTPRequest, response: HTTPResponse):
        """ë¯¸ë“¤ì›¨ì–´ ì ìš©"""
        for middleware in self.middleware:
            try:
                middleware(request, response)
            except Exception as e:
                print(f"   ë¯¸ë“¤ì›¨ì–´ ì˜¤ë¥˜: {middleware.__name__} - {e}")
