    """ìˆ«ì ë¦¬ìŠ¤íŠ¸ì˜ ê¸°ë³¸ í†µê³„ë¥¼ ê³„ì‚°í•˜ëŠ” í•¨ìˆ˜"""
    if not numbers:
        return None

    total = sum(numbers)
    count = len(numbers)
    mean = total / count

    # ì¤‘ì•™ê°’ ê³„ì‚°
    sorted_numbers = sorted(numbers)
    n = len(sorted_numbers)
    if n % 2 == 0:
        median = (sorted_numbers[n//2 - 1] + sorted_numbers[n//2]) / 2
    else:
        median = sorted_numbers[n//2]

    # ìµœë¹ˆê°’ ê³„ì‚° (ê°„ë‹¨í•œ ë°©ë²•)
    frequency = {}
    for num in numbers:
        frequency[num] = frequency.get(num, 0) + 1

    max_freq = max(frequency.values())
    mode = [num for num, freq in frequency.items() if freq == max_freq]

    return {
        "count": count,
        "sum": total,
        "mean": mean,
        "median": median,
        "mode": mode,
        "min": min(numbers),
        "max": max(numbers),
        "range": max(numbers) - min(numbers)
    }

def find_outliers(numbers, threshold=2):
    """í‰ê· ì—ì„œ ë²—ì–´ë‚œ ê°’ë“¤ì„ ì°¾ëŠ” í•¨ìˆ˜ (ê°„ë‹¨í•œ ë°©ë²•)"""
    if len(numbers) < 2:
        return []

    mean = sum(numbers) / len(numbers)

    # í‘œì¤€í¸ì°¨ ê³„ì‚°
    variance = sum((x - mean) ** 2 for x in numbers) / len(numbers)
    std_dev = variance ** 0.5

    outliers = []
    for num in numbers:
        if abs(num - mean) > threshold * std_dev:
            outliers.append(num)

    return outliers

def group_by_range(numbers, range_size=10):
    """ìˆ«ìë“¤ì„ ë²”ìœ„ë³„ë¡œ ê·¸ë£¹í™”í•˜ëŠ” í•¨ìˆ˜"""
    if not numbers:
        return {}

    min_val = min(numbers)
    max_val = max(numbers)

    groups = {}

    for num in numbers:
        # ì–´ëŠ ê·¸ë£¹ì— ì†í•˜ëŠ”ì§€ ê³„ì‚°
        group_index = int((num - min_val) // range_size)
        group_start = min_val + (group_index * range_size)
        group_end = group_start + range_size - 1

        group_key = f"{group_start}-{group_end}"

        if group_key not in groups:
            groups[group_key] = []
        groups[group_key].append(num)

    return groups
