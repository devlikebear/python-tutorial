class NumberRange:

    def __init__(self, start, end, step=1):

        self.step = step
        self.current = start

    def __iter__(self):
        """ì´í„°ëŸ¬ë¸” í”„ë¡œí† ì½œ: ìê¸° ìì‹ (ì´í„°ë ˆì´í„°)ì„ ë°˜í™˜"""

        return value
