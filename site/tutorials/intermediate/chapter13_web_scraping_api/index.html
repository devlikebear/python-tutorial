        """í…ìŠ¤íŠ¸ ì¶”ì¶œ"""

        """ëª¨ë“  ë§¤ì¹­ í…ìŠ¤íŠ¸ ì¶”ì¶œ"""

        pattern = r'<a[^>]*href=["\']([^"\']*)["\'][^>]*>(.*?)</a>'
        matches = re.findall(pattern, html, re.IGNORECASE | re.DOTALL)

    def _extract_headers(self, html):

                    'text': match.strip()

    def find(self, tag, **kwargs):

    def _rotate_user_agent(self):

    def _check_robots_txt(self, url):
        """robots.txt í™•ì¸"""
        if not self.respect_robots:
            return True

        parsed_url = urlparse(url)
        base_url = f"{parsed_url.scheme}://{parsed_url.netloc}"

        if base_url not in self.robots_cache:
            try:
                rp = RobotFileParser()
                rp.set_url(urljoin(base_url, '/robots.txt'))
                rp.read()
                self.robots_cache[base_url] = rp
            except:
                self.robots_cache[base_url] = None

        rp = self.robots_cache[base_url]
        if rp:
            return rp.can_fetch('*', url)
        return True

    def scrape_with_retry(self, url, max_retries=3):
        """ì¬ì‹œë„ ê¸°ëŠ¥ì´ ìˆëŠ” ìŠ¤í¬ë˜í•‘"""

        print(f"6. ê³ ê¸‰ ìŠ¤í¬ë˜í•‘ ê¸°ë²• ì‹œì—°:")

            print(f"  robots.txtì— ì˜í•´ ì°¨ë‹¨ëœ URL: {url}")

        for attempt in range(max_retries):
            try:
                # User-Agent ë¡œí…Œì´ì…˜
                self._rotate_user_agent()

                # ìš”ì²­ ì „ ì§€ì—°
                if attempt > 0:
                    print(f"  ì¬ì‹œë„ {attempt + 1}/{max_retries}")
                    self._random_delay()

                response.raise_for_status()

                print(f"  ì„±ê³µì ìœ¼ë¡œ í˜ì´ì§€ ê°€ì ¸ì˜´: {url}")
                print(f"  ì‚¬ìš©ëœ User-Agent: {self.session.headers.get('User-Agent')[:50]}...")

                return response.text

            except requests.exceptions.RequestException as e:
                print(f"  ì‹œë„ {attempt + 1} ì‹¤íŒ¨: {e}")
                if attempt == max_retries - 1:

                    return None

        return None
